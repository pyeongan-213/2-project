--테이블 생성
CREATE TABLE MEMBER (
MEMBER_ID number(32),
MEMBERNAME VARCHAR(255),
PASSWORD VARCHAR(255),
Birthday VARCHAR(255),
EMAIL VARCHAR(255),
JOIN_DATE VARCHAR(255),
REAL_NAME VARCHAR(255),
NICKNAME VARCHAR(255),

CONSTRAINT MEMBER_PK PRIMARY KEY(MEMBER_ID)
);

CREATE TABLE RECORDTABLE(
RECORD_ID NUMBER(32),
MEMBER_ID NUMBER(32),
MUSIC_ID NUMBER(32),
INFORMATION VARCHAR(255),
PLAYTIME NUMBER(32),
LIKE_WRITE VARCHAR(255),
LIKE_REPLY VARCHAR(255),

);

CREATE TABLE PAY(
PAY_ID NUMBER(32),

CREATE TABLE CHAT(

CREATE TABLE BOARD(
BOARD_ID NUMBER(32),

CONSTRAINT BOARD_PK PRIMARY KEY(BOARD_ID)
);

CREATE TABLE MUSIC(
MUSIC_ID NUMBER(32),
MUSIC_NAME VARCHAR(255),
ARTIST VARCHAR(255),
videoURL VARCHAR(255),
thumbnailURL VARCHAR(255),
CONSTRAINT MUSIC_PK PRIMARY KEY(MUSIC_ID)
);

--QUIZ 테이블 삭제;
CREATE TABLE quiz_music (
    quiz_id NUMBER(10) NOT NULL,         -- 퀴즈 ID
    music_id NUMBER(10) NOT NULL,        -- 음악 ID
    music_name VARCHAR2(255),            -- 노래 제목
    video_code VARCHAR2(255),            -- YouTube 비디오 코드
    start_time NUMBER(10),               -- 음악의 시작 시간 (초 단위)
    tags VARCHAR2(255),                  -- 태그 (콤마로 구분된 문자열)
    answer VARCHAR2(255),                -- 정답 (콤마로 구분된 문자열)
    CONSTRAINT quiz_music_pk PRIMARY KEY (quiz_id, music_id)
);

CREATE TABLE quiz_room (
    quiz_room_id        NUMBER GENERATED BY DEFAULT ON NULL AS IDENTITY PRIMARY KEY, -- 자동 증가 ID
    quiz_room_name     VARCHAR2(255) NOT NULL,       -- 퀴즈방 이름
    quiz_room_password VARCHAR2(255),                -- 퀴즈방 비밀번호 (NULL 허용)
    owner              VARCHAR2(255) NOT NULL,       -- 방 소유자
    status             NUMBER(1) NOT NULL,           -- 방 상태 (0: 닫힘, 1: 열림)
    member_count       NUMBER NOT NULL,              -- 현재 방의 멤버 수
    max_capacity       NUMBER NOT NULL,              -- 방의 최대 수용 인원 수
    max_music          NUMBER NOT NULL,              -- 방에서 사용할 최대 음악 곡수 
    quiz_room_type     VARCHAR(50) NOT NULL,         -- 퀴즈 타입 (예: 노래 제목 맞추기)
    version            NUMBER                        -- 낙관적 락을 위한 버전 필드
);

CREATE TABLE quiz_room_attendee (
quiz_room_member_id NUMBER GENERATED BY DEFAULT ON NULL AS IDENTITY PRIMARY KEY,
quiz_room_id NUMBER NOT NULL,
member_id NUMBER NOT NULL,
member_nickname VARCHAR2(255),
join_members VARCHAR2(1000),-- 참여 멤버 ID를 콤마로 구분하여 저장(quiz_room테이블 에서 여기로 이동)
created_at TIMESTAMP,
modified_at TIMESTAMP,
FOREIGN KEY (quiz_room_id) REFERENCES quiz_room (quiz_room_id),
FOREIGN KEY (member_id) REFERENCES member (member_id)
);


CREATE TABLE MEMBER_GAME_STATS (
    STATS_ID NUMBER(32) PRIMARY KEY,  -- 기본 키
    MEMBER_ID NUMBER(32) NOT NULL,  -- MEMBER 테이블의 외래 키
    WIN_NUM NUMBER(32) DEFAULT 0 NOT NULL,  -- 승리 횟수
    LOSE_NUM NUMBER(32) DEFAULT 0 NOT NULL,  -- 패배 횟수
    TOTAL_GAME_NUM NUMBER(32) DEFAULT 0 NOT NULL,  -- 총 게임 횟수
    ENTER_GAME_NUM NUMBER(32) DEFAULT 0 NOT NULL,  -- 게임 참여 횟수
    SOLO_EXIT_NUM NUMBER(32) DEFAULT 0 NOT NULL,  -- 단독 퇴장 횟수
    MAKE_ROOM_NUM NUMBER(32) DEFAULT 0 NOT NULL,  -- 방 생성 횟수
    PLAY_TIME NUMBER(32) DEFAULT 0 NOT NULL,  -- 플레이 시간
    score NUMBER(32) DEFAULT 0 NOT NULL,
    CONSTRAINT FK_MEMBER_GAME_STATS_MEMBER FOREIGN KEY (MEMBER_ID) REFERENCES MEMBER (MEMBER_ID)
);
